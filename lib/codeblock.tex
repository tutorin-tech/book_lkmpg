\NewDocumentCommand{\samplec}{oom}{%
  \IfNoValueTF{#1}%
  {%
    \VerbatimInput[
      frame=single,
      framesep=2mm,
      baselinestretch=1,
      fontsize=\footnotesize,
      breaklines=true,
      numbers=left
    ]{#3}%
  }%
  {%
    \IfNoValueTF{#2}%
    {%
      \VerbatimInput[
        frame=single,
        framesep=2mm,
        baselinestretch=1,
        fontsize=\footnotesize,
        breaklines=true,
        firstline=#1,
        numbers=left
      ]{#3}%
    }%
    {%
      \VerbatimInput[
        frame=single,
        framesep=2mm,
        baselinestretch=1,
        fontsize=\footnotesize,
        breaklines=true,
        firstline=#1,
        lastline=#2,
        numbers=left
      ]{#3}%
    }%
  }%
}
